Identity function

\( id_D(d) = d \)

idD = lambda d: d

def is_identity_function(idD, D):
    for d in D:
        if idD(d) != d:
            return false
    return true

Function composition

\( f: A \rightarrow B, \quad g: B \rightarrow C, \quad g \circ f: A \rightarrow C, \quad (g \circ f)(x) = g(f(x)) \)

def f(x: A) -> B:
    ...
    return y

def g(x: B) -> C:
    ...
    return y

def composite_function(g, f):
    def g_o_f(x):
        return g(f(x))
    return g_o_f

composite_function(g, f)(x) == g(f(x))

One to one (injective) function

 
# for f: D -> F
f: lambda x: ...
D = {...}
F = {...}
def is_one_to_one(f, D):
    for x in D:
        for y in D:
            if f(x) == f(y) and x != y:
                return false
    return true

Onto (surjective) function

 
def is_onto(f, D, F):
    _range = set(map(f, D))
    for z in F:
        if z not in _range:
            return false
    return true

Bijective function

 
def is_bijective(f, D, F):
    return is_one_to_one(f, D) and is_onto(f, D, F)

Invertible function

 
def is_invertible(f, D, F):
    return is_bijective(f, D, F)
